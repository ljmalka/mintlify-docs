export const makeIntegration = ({
id, name, vertical, kind, status,
auth, // { type: 'API_KEY'|'OAUTH'|'NONE' }
events, // { inbound: string[], outbound: string[] }
features, // object of flags/strings
privacy, // { pii, consent, dataResidency }
limits, // { rateLimit, pagination, webhooks }
docsLinks, // { vendor, apiRef, status, pricing }
plans, // [ 'Starter'|'Pro'|'Enterprise' ]
notes // optional string[]
}) => ({
id, name, vertical, kind, status,
auth, events, features, privacy, limits, docsLinks, plans, notes,
});
